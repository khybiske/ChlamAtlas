> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Mutants two-panel UI — nav dropdown entry, collection-filtered list with search/sort/filter, and a dashboard-style detail panel with gene cards, pipeline, phenotypes, and stocks.

**Architecture:** Full rewrite of `mutants.js` as a two-panel layout matching the Genomes tab structure (list-panel left, detail-panel right). Detail panel renders white cards on a gray background, distinct from the gene detail scroll style.

**Tech Stack:** Vanilla JS ES modules, Supabase JS v2 client (`sb`), Tailwind CSS (CDN), custom CSS in `app.css`. No new dependencies.

## Task 1: Schema Migration 021

**Files:**

- [ ] **Step 1: Create the migration file**

```sql
-- supabase/migrations/021_mutants_schema.sql
-- Rename mutation_type 'recombination' → 'deletion', remove 'intron', add mutation_method

-- 1. Drop the existing check constraint
ALTER TABLE public.mutants
  DROP CONSTRAINT IF EXISTS mutants_mutation_type_check;

-- 2. Backfill: recombination → deletion, intron → deletion
UPDATE public.mutants
  SET mutation_type = 'deletion'
  WHERE mutation_type IN ('recombination', 'intron');

-- 3. Add new constraint with updated enum
ALTER TABLE public.mutants
  ADD CONSTRAINT mutants_mutation_type_check
  CHECK (mutation_type IN ('transposon', 'deletion', 'chemical'));

-- 4. Add mutation_method for deletion subtypes
ALTER TABLE public.mutants
  ADD COLUMN IF NOT EXISTS mutation_method text;
-- Values: 'lambda_red', 'targetron', 'crispr', NULL (for transposon/chemical)

COMMENT ON COLUMN public.mutants.mutation_method IS
  'Deletion subtype: lambda_red, targetron, crispr, or NULL for non-deletion types';
```

// ── Fetch mutant list ─────────────────────────────────────
async function fetchList(container) {
  const scroll = container.querySelector('#mut-scroll');
  if (!scroll) return;
  scroll.innerHTML = skeletonRows(6);

  const from = _page * PAGE_SIZE;
  const to   = from + PAGE_SIZE - 1;

  let query = sb
    .from('mutants')
    .select('id, mutant_id, name, mutation_type, is_published, collection', { count: 'exact' })
    .eq('collection', _collection)
    .order(_sortCol, { ascending: _sortAsc })
    .range(from, to);

  if (_typeFilter)  query = query.eq('mutation_type', _typeFilter);
  if (_searchTerm)  query = query.or(`mutant_id.ilike.%${_searchTerm}%,name.ilike.%${_searchTerm}%`);

  // Published filter: public users only see published (RLS enforces this,
  // but explicit filter lets lab members browse unpublished-only)
  if (_activeFilters.published === 'true')  query = query.eq('is_published', true);
  if (_activeFilters.published === 'false') query = query.eq('is_published', false);

  const { data, count, error } = await query;
  _total = count ?? 0;

  if (error) {
    scroll.innerHTML = `<div style="padding:20px;color:#ef4444;font-size:12px;">${error.message}</div>`;
    return;
  }
  if (!data?.length) {
    scroll.innerHTML = `<div style="padding:32px;text-align:center;color:#9ca3af;font-size:13px;">No mutants found.</div>`;
    renderPager(container);
    return;
  }

  scroll.innerHTML = data.map(m => mutantRowHTML(m)).join('');
  scroll.querySelectorAll('.mut-row').forEach(row => {
    row.addEventListener('click', () => selectMutant(row.dataset.id, container));
  });

  // Update strip count
  const countEl = container.querySelector('#mut-strip-count');
  if (countEl) countEl.textContent = `${_total.toLocaleString()} mutants`;

  // Auto-select first if nothing selected
  if (!_selectedId && data.length) selectMutant(data[0].id, container);
  else if (_selectedId) highlightSelected(container);

  renderPager(container);
}

    detailEl.innerHTML = `<div style="padding:20px;color:#ef4444;font-size:12px;">Mutant not found.</div>`;
    return;
  }

  // Resolve target genes (UUID[]) → gene records with AF3 thumbnail
  let targetGenes = [];
  if (m.target_gene_ids?.length) {
    const { data: genes } = await sb
      .from('genes')
      .select(`
        id, locus_tag, gene_name, protein_product, functional_category, is_characterized,
        proteins(alphafold_results(thumbnail_path))
      `)
      .in('id', m.target_gene_ids);
    targetGenes = genes ?? [];
  }

  const displayName = m.name || m.mutant_id;
  const strainLabel = m.strain?.common_name ?? m.collection ?? '';
  const typeLabel   = { transposon: 'Transposon', deletion: 'Deletion', chemical: 'Chemical' }[m.mutation_type] ?? m.mutation_type ?? '';
  const creatorName = m.creator?.display_name ?? '—';
  const isLabMember = ['lab_member', 'admin'].includes(state.userRole);

  detailEl.innerHTML = `
    ${heroHTML(displayName, m.mutant_id, strainLabel, typeLabel, m.is_published)}
    ${targetGenes.length ? geneCardsHTML(targetGenes) : ''}
    ${recombInfoHTML(m, pipe, creatorName)}
    ${pipelineHTML(pipe, isLabMember)}
    ${phenoHTML(phenos)}
    ${stocksHTML(pipe, isLabMember)}
    <div style="height:24px;"></div>`;

  // Wire gene card links → Genomes tab
  detailEl.querySelectorAll('[data-gene-id]').forEach(btn => {
    btn.addEventListener('click', () => {
      window.__geneDetailId    = btn.dataset.geneId;
      window.__preferredStrain = btn.dataset.strain ?? null;
      document.querySelector('[data-tab="genomes"]').click();
    });
  });
}

function geneCardsHTML(genes) {
  const cards = genes.map(g => {

          ${catBadge ? `<div style="margin-top:4px;">${catBadge}</div>` : ''}
        </div>
        <div class="mut-gene-link">View in Genomes →</div>
      </div>`;
  }).join('<div style="height:6px;"></div>');

    </div>`;
}

// Functional category badge (matches gene detail colour scheme)
function funcBadge(cat, isCharacterized) {
  if (!cat && !isCharacterized) return '';
  const map = {
    't3ss':     { bg:'#ccfbf1', text:'#0f766e', label:'T3SS' },
    'inc':      { bg:'#ede9fe', text:'#5b21b6', label:'Inc protein' },
    'division': { bg:'#fef3c7', text:'#92400e', label:'Cell division' },
    'reg':      { bg:'#fee2e2', text:'#991b1b', label:'Regulatory' },
  };
  const s = map[cat];
  if (s) return `<span style="font-size:9px;font-weight:600;text-transform:uppercase;letter-spacing:.04em;padding:2px 6px;border-radius:6px;background:${s.bg};color:${s.text};">${s.label}</span>`;
  if (isCharacterized) return `<span style="font-size:9px;font-weight:600;text-transform:uppercase;letter-spacing:.04em;padding:2px 6px;border-radius:6px;background:#f3f4f6;color:#6b7280;">Characterized</span>`;
  return `<span style="font-size:9px;font-weight:600;text-transform:uppercase;letter-spacing:.04em;padding:2px 6px;border-radius:6px;background:#f3f4f6;color:#9ca3af;">Hypothetical</span>`;
}
